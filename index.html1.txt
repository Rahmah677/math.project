    <script>
        // Crop Growth Constants
        const L = 150;      // Carrying capacity (g/hill)
        const k = 0.08;     // Growth rate coefficient
        const x0 = 55;      // Inflection point (Days After Transplanting)

        // Mathematical formulation functions
        function getBiomass(x) {
            return L / (1 + Math.exp(-k * (x - x0)));
        }

        function getGrowthRate(x) {
            const expTerm = Math.exp(-k * (x - x0));
            return (L * k * expTerm) / Math.pow(1 + expTerm, 2);
        }

        // Generate data points for Chart.js
        const labels = [];
        const biomassData = [];
        const rateData = [];

        for (let i = 0; i <= 110; i += 2) {
            labels.push(i);
            biomassData.push(getBiomass(i));
            rateData.push(getGrowthRate(i));
        }

        // DOM elements
        const daySlider = document.getElementById('daySlider');
        const sliderLabel = document.getElementById('sliderLabel');
        const currentDaySpan = document.getElementById('currentDay');
        const currentBiomassSpan = document.getElementById('currentBiomass');
        const currentRateSpan = document.getElementById('currentRate');
        const stageBadge = document.getElementById('stageBadge');
        const stageTitle = document.getElementById('stageTitle');
        const laymanExplanation = document.getElementById('laymanExplanation');

        // Physiological Stages Data for MR297 Variety
        function getStageAnalysis(day, rate, biomass) {
            if (day < 25) {
                return {
                    badge: "EARLY VEGETATIVE",
                    title: "Seedling Establishment & Early Tillering",
                    explanation: "The newly transplanted seedlings are focusing on root establishment. The crop growth rate is relatively low (" + rate.toFixed(2) + " g/day). Keep weed populations low and soil moist, but heavy fertilizers are not yet needed as uptake capacity is minimal.",
                    colorClass: "from-stone-900 to-emerald-950 border-emerald-500"
                };
            } else if (day >= 25 && day < 50) {
                return {
                    badge: "VEGETATIVE SURGE",
                    title: "Stem Elongation & Rapid Tillering",
                    explanation: "Growth is accelerating exponentially! The growth rate is ramping up rapidly. This is the optimal window for applying Nitrogen-rich fertilizer (Urea) because the plant has strong leaf-transpiration suction to absorb nutrients and produce maximum productive tillers.",
                    colorClass: "from-emerald-900 to-teal-950 border-teal-500"
                };
            } else if (day >= 50 && day < 65) {
                return {
                    badge: "PEAK INFLECTION",
                    title: "Panicle Initiation & Heading Preparation",
                    explanation: "This is the mathematical inflection point! Growth rate peaks at " + rate.toFixed(2) + " g/day. The plant is transitioning energy from producing new leaves to forming the actual grain structure. Timing water and micro-nutrient inputs right now is critical for final yield output.",
                    colorClass: "from-amber-950 to-emerald-950 border-amber-500"
                };
            } else if (day >= 65 && day < 85) {
                return {
                    badge: "REPRODUCTIVE PHASE",
                    title: "Flowering & Grain Development",
                    explanation: "Cumulative biomass is nearing plateau (" + biomass.toFixed(1) + " g). The daily growth rate is declining because the structural architecture of the rice is finished. The plant is now concentrating on grain filling, sending sugars directly into the developing panicles.",
                    colorClass: "from-amber-900 to-stone-900 border-amber-600"
                };
            } else {
                return {
                    badge: "MATURITY & HARVEST",
                    title: "Grain Ripening & Yellowing",
                    explanation: "The plant growth rate has dropped near zero (" + rate.toFixed(2) + " g/day). This indicates complete physiological maturity of the MR297 grain. Field drainage should begin now to dry the grains for modern mechanical harvesting, securing Malaysian food supply!",
                    colorClass: "from-yellow-950 to-stone-900 border-yellow-600"
                };
            }
        }

        // Generate dynamic Tangent Line Dataset around selected day
        function getTangentDataset(selectedDay) {
            const y1 = getBiomass(selectedDay);
            const slope = getGrowthRate(selectedDay);
            const tangentPoints = [];
            
            // Generate local line segment of length 20 days centered at selectedDay
            const range = 20;
            const startX = Math.max(0, selectedDay - range);
            const endX = Math.min(110, selectedDay + range);

            for (let x = startX; x <= endX; x += 1) {
                // Point-slope form: y - y1 = m(x - x1)  =>  y = m(x - x1) + y1
                const y = slope * (x - selectedDay) + y1;
                tangentPoints.push({ x: x, y: y });
            }
            return tangentPoints;
        }

        // Initialize Chart
        const ctx = document.getElementById('growthChart').getContext('2d');
        let initialSelectedDay = 55;
        
        const growthChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [
                    {
                        label: 'Biomass Accumulation f(x) (g/hill)',
                        data: biomassData,
                        borderColor: '#059669', // Emerald 600
                        backgroundColor: 'rgba(5, 150, 105, 0.05)',
                        borderWidth: 3,
                        pointRadius: 0,
                        tension: 0.1,
                        yAxisID: 'y'
                    },
                    {
                        label: 'Daily Growth Rate f\'(x) (g/day)',
                        data: rateData,
                        borderColor: 'rgba(217, 119, 6, 0.4)', // Amber 600, semi-transparent
                        backgroundColor: 'transparent',
                        borderWidth: 2,
                        borderDash: [5, 5],
                        pointRadius: 0,
                        tension: 0.1,
                        yAxisID: 'y'
                    },
                    {
                        label: 'Tangent Line (Local Slope)',
                        data: getTangentDataset(initialSelectedDay),
                        borderColor: '#ef4444', // Red 500
                        backgroundColor: 'transparent',
                        borderWidth: 2.5,
                        pointRadius: 0,
                        fill: false,
                        showLine: true,
                        yAxisID: 'y'
                    },
                    {
                        label: 'Selected Day Indicator',
                        data: [{ x: initialSelectedDay, y: getBiomass(initialSelectedDay) }],
                        borderColor: '#ef4444',
                        backgroundColor: '#ef4444',
                        pointRadius: 7,
                        pointHoverRadius: 9,
                        showLine: false,
                        yAxisID: 'y'
                    }
                ]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                interaction: {
                    mode: 'index',
                    intersect: false
                },
                plugins: {
                    legend: {
                        display: false // We use custom header legend to save canvas space
                    },
                    tooltip: {
                        enabled: false // Custom text readout handles this beautifully
                    }
                },
                scales: {
                    x: {
                        type: 'linear',
                        title: {
                            display: true,
                            text: 'Days After Transplanting (DAT)',
                            color: '#1c1917',
                            font: {
                                weight: 'bold',
                                size: 12
                            }
                        },
                        ticks: {
                            color: '#44403c'
                        },
                        grid: {
                            color: '#f5f5f4'
                        },
                        min: 0,
                        max: 110
                    },
                    y: {
                        type: 'linear',
                        display: true,
                        position: 'left',
                        title: {
                            display: true,
                            text: 'Padi Dry Weight Biomass (g/hill)',
                            color: '#1c1917',
                            font: {
                                weight: 'bold',
                                size: 12
                            }
                        },
                        ticks: {
                            color: '#44403c'
                        },
                        grid: {
                            color: '#f5f5f4'
                        },
                        min: 0,
                        max: 170
                    }
                },
                onClick: (e) => {
                    const canvasPosition = Chart.helpers.getRelativePosition(e, growthChart);
                    const dataX = growthChart.scales.x.getValueForPixel(canvasPosition.x);
                    
                    if (dataX >= 0 && dataX <= 110) {
                        const targetDay = Math.round(dataX);
                        daySlider.value = targetDay;
                        updateVisualization(targetDay);
                    }
                }
            }
        });

        // Core interactive update logic
        function updateVisualization(day) {
            // Read formula values at selected Day
            const biomass = getBiomass(day);
            const rate = getGrowthRate(day);

            // Update DOM text readouts
            sliderLabel.textContent = `Day ${day}`;
            currentDaySpan.textContent = day;
            currentBiomassSpan.textContent = biomass.toFixed(1);
            currentRateSpan.textContent = (rate >= 0 ? "+" : "") + rate.toFixed(2);

            // Get bioscience agricultural analysis & change color of active banner dynamically
            const analysis = getStageAnalysis(day, rate, biomass);
            stageBadge.textContent = analysis.badge;
            stageTitle.textContent = analysis.title;
            laymanExplanation.innerHTML = analysis.explanation;
            
            // Adjust card background gradient based on growth phase
            const card = document.getElementById('analysisCard');
            card.className = `bg-gradient-to-br ${analysis.colorClass} text-stone-100 rounded-xl p-6 shadow-md border-b-4 transition-all duration-300 transform`;

            // Update chart datasets
            growthChart.data.datasets[2].data = getTangentDataset(day); // Rebuild tangent line
            growthChart.data.datasets[3].data = [{ x: day, y: biomass }]; // Move indicator dot
            growthChart.update('none'); // Animate smoothly without full reset
        }

        // Slider input listener
        daySlider.addEventListener('input', (e) => {
            updateVisualization(parseInt(e.target.value));
        });

        // Setup MathJax/KaTeX math formula rendering on page load
        function renderMath() {
            if (typeof katex !== 'undefined') {
                katex.render("f(x) = \\frac{150}{1 + e^{-0.08(x - 55)}}", document.getElementById('math-fx'), {
                    throwOnError: false
                });
                katex.render("f'(x) = \\frac{12 e^{-0.08(x - 55)}}{\\left(1 + e^{-0.08(x - 55)}\\right)^2}", document.getElementById('math-dfx'), {
                    throwOnError: false
                });
            }
        }

        // Initialize state on window load
        window.onload = function() {
            updateVisualization(55);
            renderMath();
        };
    </script>
